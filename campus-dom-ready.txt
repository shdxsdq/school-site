    window.addEventListener('error', (event) => {
      const hint = document.querySelector('.campus-stage-bottom p');
      const description = document.querySelector('#spot-description');
      const source = event.filename ? `（${event.filename.split('/').pop()}）` : '';
      const message = event.message ? `${event.message}${source}` : '三维导览初始化失败';

      if (hint) {
        hint.textContent = `三维导览初始化失败：${message}`;
      }

      if (description) {
        description.textContent = `三维导览初始化失败：${message}`;
      }
    });

    window.addEventListener('unhandledrejection', (event) => {
      const hint = document.querySelector('.campus-stage-bottom p');
      const description = document.querySelector('#spot-description');
      const reason = event.reason && event.reason.message ? event.reason.message : '模块脚本未能完成初始化';

      if (hint) {
        hint.textContent = `三维导览初始化失败：${reason}`;
      }

      if (description) {
        description.textContent = `三维导览初始化失败：${reason}`;
      }
    });
  </script>
